Sample Finding

Real bug I found in a production codebase

// File: src/api/auth.js — Line 47
async function verifyToken(token) {
  const decoded = jwt.verify(token, process.env.JWT_SECRET);
  // BUG: No check that 'decoded' is not null/undefined
  return decoded.userId;  // 💀 throws if token is malformed
}

// IMPACT:
• Malformed token → 500 Internal Server Error
• Attackers can probe to discover token structure
• No rate limiting on auth endpoint

// FIX:
if (!decoded || !decoded.userId) {
  throw new AuthError('Invalid token', 401);
}
